\begin{tcolorbox}[colback=celesteSuave, colframe=azulFuerte, title=\textbf{RESUMEN DE CLASES: LO QUE DEBES SABER}, fonttitle=\bfseries]
\begin{multicols}{2}
    \textbf{1. Razones en el Triángulo}
    \begin{itemize}[leftmargin=10pt]
        \item $\sin \alpha = \frac{\text{Opuesto}}{\text{Hipotenusa}}$ \quad $\csc \alpha = \frac{1}{\sin \alpha}$
        \item $\cos \alpha = \frac{\text{Adyacente}}{\text{Hipotenusa}}$ \quad $\sec \alpha = \frac{1}{\cos \alpha}$
        \item $\tan \alpha = \frac{\text{Opuesto}}{\text{Adyacente}}$ \quad $\cot \alpha = \frac{1}{\tan \alpha}$
    \end{itemize}

    \columnbreak

    \textbf{2. Identidades Fundamentales}
    \begin{itemize}[leftmargin=10pt]
        \item \textbf{Pitágoras I:} $\sin^2 \alpha + \cos^2 \alpha = 1$
        \item \textbf{Pitágoras II:} $1 + \tan^2 \alpha = \sec^2 \alpha$
        \item \textbf{Pitágoras III:} $1 + \cot^2 \alpha = \csc^2 \alpha$
        \item \textbf{Cociente:} $\tan \alpha = \dfrac{\sin \alpha}{\cos \alpha}$ \quad $\cot \alpha = \dfrac{\cos \alpha}{\sin \alpha}$
    \end{itemize}
\end{multicols}

\vspace{0.1cm}
\textbf{3. Identidades de Suma y Resta de Ángulos}
\begin{itemize}[leftmargin=10pt, itemsep=0.15cm]
    \item \textbf{Suma Seno} (\textit{sen-cos-cos-sen}): $\sin(\alpha \pm \beta) = \sin \alpha \cos \beta \pm \cos \alpha \sin \beta$
    \item \textbf{Suma Coseno} (\textit{cos-cos-sen-sen}): $\cos(\alpha \pm \beta) = \cos \alpha \cos \beta \mp \sin \alpha \sin \beta$
\end{itemize}

\vspace{0.1cm}
\textbf{4. Identidades de Ángulo Doble}
\begin{itemize}[leftmargin=10pt, itemsep=0.15cm]
    \item $\sin(2\alpha) = 2\sin\alpha\cos\alpha$
    \item $\cos(2\alpha) = \cos^2\alpha - \sin^2\alpha = 2\cos^2\alpha - 1 = 1 - 2\sin^2\alpha$
\end{itemize}

\vspace{0.1cm}
\textbf{5. Identidades Útiles de Simplificación}
\begin{itemize}[leftmargin=10pt, itemsep=0.15cm]
    \item $\sin\alpha \cdot \csc\alpha = 1$ \qquad $\cos\alpha \cdot \sec\alpha = 1$ \qquad $\tan\alpha \cdot \cot\alpha = 1$
    \item $\tan \alpha + \cot \alpha = \sec\alpha\,\csc\alpha$
\end{itemize}

\vspace{-0.1cm}
\begin{center}
    \small \textbf{Tip Pro:} Si $\alpha + \beta = 90^\circ$ (complementarios), entonces $\sin \alpha = \cos \beta$.
\end{center}
